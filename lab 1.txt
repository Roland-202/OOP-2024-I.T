encryption 

step 1
# Create working folder
mkdir simple_gpg
cd simple_gpg

# Create a secret file
echo "This is my top secret message!" > secret.txt
cat secret.txt

step 2
# Generate a simple key (use simple passwords for this demo)
gpg --batch --generate-key << EOF
Key-Type: RSA
Key-Length: 2048
Name-Real: Test User
Name-Email: test@example.com
Expire-Date: 0
Passphrase: 1234
%commit
EOF

# Check if key was created
gpg --list-keys

step 3
# Encrypt using your key
gpg --encrypt --recipient "Test User" secret.txt

# You should see: secret.txt.gpg created
ls -la

decrypt step 4
# Decrypt the file
gpg --decrypt secret.txt.gpg > decrypted.txt

# Enter passphrase: 1234

encrypt with key
gpg --encrypt --recipient "Test User" message.txt

decrypt with key
